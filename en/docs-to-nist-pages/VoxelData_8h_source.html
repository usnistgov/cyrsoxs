76  void addMaterialDataVectorMorphology(py::array_t<Real, py::array::c_style | py::array::forcecast> &matAlignementData,
77  py::array_t<Real, py::array::c_style | py::array::forcecast> &matUnalignedData,
78  const UINT matID) {
79  const int NUM_MATERIAL = inputData_.NUM_MATERIAL;
80  if (inputData_.morphologyType == MorphologyType::EULER_ANGLES) {
81  py::print("Error: [Expected]: Vector Morphology [Found:] Euler Angles. Returning\n");
82  return;
83  }
84  if (matID > NUM_MATERIAL or (matID == 0)) {
85  throw std::logic_error(
86  "Number of material does not match with the compiled version. matID must range from 1 to NUM_MATERIAL");
87  }
88  if (validData_.test(matID - 1)) {
89  py::print("The material is already set. Please first reset to add the entries. Returning.");
90  return;
91  }
92  const BigUINT numVoxels = inputData_.voxelDims[0] * inputData_.voxelDims[1] * inputData_.voxelDims[2];
93  std::vector<Real> _matAlignedData(numVoxels * 3);
94  std::vector<Real> _matUnalignedData(numVoxels);
95  for (int i = 0; i < numVoxels; i++) {
96  _matAlignedData[3 * i + 0] = matAlignementData.data()[3 * i + 0];
97  _matAlignedData[3 * i + 1] = matAlignementData.data()[3 * i + 1];
98  _matAlignedData[3 * i + 2] = matAlignementData.data()[3 * i + 2];
99  _matUnalignedData[i] = matUnalignedData.data()[i];
100  }
101  if (inputData_.morphologyOrder == MorphologyOrder::XYZ) {
102  H5::XYZ_to_ZYX(_matUnalignedData, 1, inputData_.voxelDims);
103  H5::XYZ_to_ZYX(_matAlignedData, 3, inputData_.voxelDims);
104  }
105  for (BigUINT i = 0; i < numVoxels; i++) {
106  voxel[(matID - 1) * numVoxels + i].s1.x = _matAlignedData.data()[i * 3 + 0];
107  voxel[(matID - 1) * numVoxels + i].s1.y = _matAlignedData.data()[i * 3 + 1];
108  voxel[(matID - 1) * numVoxels + i].s1.z = _matAlignedData.data()[i * 3 + 2];
109  voxel[(matID - 1) * numVoxels + i].s1.w = _matUnalignedData.data()[i];
110  }
111  validData_.set(matID - 1, true);
112  }

122  void addMaterialDataEulerAngles(py::array_t<Real, py::array::c_style | py::array::forcecast> &matSVector,
123  py::array_t<Real, py::array::c_style | py::array::forcecast> &matThetaVector,
124  py::array_t<Real, py::array::c_style | py::array::forcecast> &matPsiVector,
125  py::array_t<Real, py::array::c_style | py::array::forcecast> &matVfracVector,
126  const UINT matID) {
127  const int NUM_MATERIAL = inputData_.NUM_MATERIAL;
128 
129  if (inputData_.morphologyType != MorphologyType::EULER_ANGLES) {
130  py::print("Error: [Expected]: Euler Angles / Spherical Coordinates. [Found:] VectorMorphology\n");
131  return;
132  }
133  if ((matID > NUM_MATERIAL) or (matID == 0)) {
134  throw std::logic_error(
135  "Number of material does not match with the compiled version. matID must range from 1 to NUM_MATERIAL");
136  }
137  if (validData_.test(matID - 1)) {
138  py::print("The material is already set. Please first reset to add the entries. Returning.");
139  return;
140  }
141  const BigUINT numVoxels = inputData_.voxelDims[0] * inputData_.voxelDims[1] * inputData_.voxelDims[2];
142  std::vector<Real> _S(numVoxels);
143  std::vector<Real> _Theta(numVoxels);
144  std::vector<Real> _Psi(numVoxels);
145  std::vector<Real> _Vfrac(numVoxels);
146  for (BigUINT i = 0; i < numVoxels; i++) {
147  _S[i] = matSVector.data()[i];
148  _Theta[i] = matThetaVector.data()[i];
149  _Psi[i] = matPsiVector.data()[i];
150  _Vfrac[i] = matVfracVector.data()[i];
151  }
152  if (inputData_.morphologyOrder == MorphologyOrder::XYZ) {
153  H5::XYZ_to_ZYX(_S, 1, inputData_.voxelDims);
154  H5::XYZ_to_ZYX(_Theta, 1, inputData_.voxelDims);
155  H5::XYZ_to_ZYX(_Psi, 1, inputData_.voxelDims);
156  H5::XYZ_to_ZYX(_Vfrac, 1, inputData_.voxelDims);
157  }
158  for (BigUINT i = 0; i < numVoxels; i++) {
159  voxel[(matID - 1) * numVoxels + i].s1.x = _S[i];
160  if (_S[i] != 0) {
161  voxel[(matID - 1) * numVoxels + i].s1.y = _Theta[i];
162  voxel[(matID - 1) * numVoxels + i].s1.z = _Psi[i];
163  } else {
164  voxel[(matID - 1) * numVoxels + i].s1.y = 0;
165  voxel[(matID - 1) * numVoxels + i].s1.z = 0;
166  }
167  voxel[(matID - 1) * numVoxels + i].s1.w = _Vfrac[i];
168  }
169 
170  validData_.set(matID - 1, true);
171  }

217  void readFromH5(const std::string &fname) {
218  if (validData_.any()) {
219  py::print("Some of the material is already set. Please first reset to continue. Returning.");
220  return;
221  }
222  int numMaterial = H5::getNumberOfMaterial(fname);
223  if(numMaterial != inputData_.NUM_MATERIAL){
224  py::print("[HDF5 Error]: Number of material.");
225  return;
226  }
227 
228  MorphologyOrder morphologyOrder;
229  Real physSize;
230  UINT numVoxels[3];
231  H5::getDimensionAndOrder(fname,(MorphologyType)inputData_.morphologyType,numVoxels,physSize,morphologyOrder);
232  if(inputData_.morphologyOrder != morphologyOrder){
233  py::print("[HDF5 Error]: Morphology Order mismatch.");
234  return;
235  }
236 
237  if(!(FEQUALS(inputData_.physSize,physSize))){
238  py::print("[HDF5 Error]: PhysSize mismatch.");
239  return;
240  }
241 
242  H5::readFile(fname, inputData_.voxelDims, voxel, (MorphologyType) inputData_.morphologyType,
243  inputData_.morphologyOrder, inputData_.NUM_MATERIAL,true);
244  validData_.set();
245  }
