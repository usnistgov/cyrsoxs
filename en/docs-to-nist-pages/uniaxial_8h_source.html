59 __device__ void computePolarizationEulerAngles(const Material *material,
60  const Voxel *voxelInput, const BigUINT threadID,
61  Complex *polarizationX, Complex *polarizationY, Complex *polarizationZ,
62  const BigUINT & numVoxels, const Matrix & rotationMatrix, int NUM_MATERIAL) {

64  Complex pX{0.0,0.0}, pY{0.0,0.0}, pZ{0.0,0.0};

66  static constexpr Real OneBy4Pi = static_cast<Real> (1.0 / (4.0 * M_PI));
72  Complex rotatedNr{0.0,0.0}; // Only storing what is required
73  static constexpr Real3 eleField{1,0,0};
74  Real3 matVec;
75  doMatVec<false>(rotationMatrix,eleField,matVec);
76  Complex nsum;
77  for (int numMaterial = 0; numMaterial < NUM_MATERIAL; numMaterial++) {
78  Complex npar = material[numMaterial].npara;
79  Complex nper = material[numMaterial].nperp;
80  const Voxel matProp = voxelInput[numVoxels * numMaterial + threadID];
81 
82  const Real & psiAngle = matProp.getValueAt(Voxel::EULER_ANGLE::PSI);
83  const Real & thetaAngle = matProp.getValueAt(Voxel::EULER_ANGLE::THETA);
84  const Real & Vfrac = matProp.getValueAt(Voxel::EULER_ANGLE::VFRAC);
85  const Real & phi_a = Vfrac*matProp.getValueAt(Voxel::EULER_ANGLE::S);
86 
87  const Real sx = cos(psiAngle)*sin(thetaAngle);
88  const Real sy = sin(psiAngle)*sin(thetaAngle);
89  const Real sz = cos(thetaAngle);
90  const Real phi_ui = Vfrac - phi_a;
91 
92  const Real & phi = Vfrac;
93 
94  nsum.x = npar.x + 2 * nper.x;
95  nsum.y = npar.y + 2 * nper.y;
96 
97  computeComplexSquare(nsum);
98  computeComplexSquare(npar);
99  computeComplexSquare(nper);
100 
101  // S = fraction of aligned components
102  // (0)
103  rotatedNr.x = phi_a*(npar.x*sx*sx + nper.x*(sy*sy + sz*sz)) + ((phi_ui * nsum.x) / (Real) 9.0) - phi;
104  rotatedNr.y = phi_a*(npar.y*sx*sx + nper.y*(sy*sy + sz*sz)) + ((phi_ui * nsum.y) / (Real) 9.0);
105 
106  pX.x += rotatedNr.x*matVec.x;
107  pX.y += rotatedNr.y*matVec.x;
108 
109  // (1)
110  rotatedNr.x = phi_a*(npar.x - nper.x)*sx*sy;
111  rotatedNr.y = phi_a*(npar.y - nper.y)*sx*sy;
112 
113  pX.x += rotatedNr.x*matVec.y;
114  pX.y += rotatedNr.y*matVec.y;
115 
116  pY.x += rotatedNr.x*matVec.x;
117  pY.y += rotatedNr.y*matVec.x;
118 
119  // (2)
120  rotatedNr.x = phi_a*(npar.x - nper.x)*sx*sz;
121  rotatedNr.y = phi_a*(npar.y - nper.y)*sx*sz;
122 
123  pX.x += rotatedNr.x*matVec.z;
124  pX.y += rotatedNr.y*matVec.z;
125 
126  pZ.x += rotatedNr.x*matVec.x;
127  pZ.y += rotatedNr.y*matVec.x;
128 
129  // (3)
130  rotatedNr.x = phi_a*(npar.x*sy*sy + nper.x*(sx*sx + sz*sz)) + ((phi_ui * nsum.x) / (Real) 9.0) - phi;
131  rotatedNr.y = phi_a*(npar.y*sy*sy + nper.y*(sx*sx + sz*sz)) + ((phi_ui * nsum.y) / (Real) 9.0);
132 
133  pY.x += rotatedNr.x*matVec.y;
134  pY.y += rotatedNr.y*matVec.y;
135 
136  // (4)
137  rotatedNr.x = phi_a*(npar.x - nper.x)*sy*sz;
138  rotatedNr.y = phi_a*(npar.y - nper.y)*sy*sz;
139 
140  pY.x += rotatedNr.x*matVec.z;
141  pY.y += rotatedNr.y*matVec.z;
142 
143  pZ.x += rotatedNr.x*matVec.y;
144  pZ.y += rotatedNr.y*matVec.y;
145 
146  // (5)
147  rotatedNr.x = phi_a*(npar.x*sz*sz + nper.x*(sx*sx + sy*sy)) + ((phi_ui * nsum.x) / (Real) 9.0) - phi;
148  rotatedNr.y = phi_a*(npar.y*sz*sz + nper.y*(sx*sx + sy*sy)) + ((phi_ui * nsum.y) / (Real) 9.0);
149 
150  pZ.x += rotatedNr.x*matVec.z;
151  pZ.y += rotatedNr.y*matVec.z;
152 
153  }
154 
155  pX.x *= OneBy4Pi;
156  pX.y *= OneBy4Pi;
157 
158  pY.x *= OneBy4Pi;
159  pY.y *= OneBy4Pi;
160 
161  pZ.x *= OneBy4Pi;
162  pZ.y *= OneBy4Pi;
163  if(referenceFrame == ReferenceFrame::MATERIAL) {
164  rotate<true>(rotationMatrix,pX,pY,pZ);
165  }
166  polarizationX[threadID] = pX;
167  polarizationY[threadID] = pY;
168  polarizationZ[threadID] = pZ;
