164  template<typename T>
165  static void XYZ_to_ZYX(std::vector<T> &data, const int numComponents, const UINT *voxelSize) {
166  std::vector<T> _data = data;
167  int counter = 0;
168  UINT X = voxelSize[0];
169  UINT Y = voxelSize[1];
170  UINT Z = voxelSize[2];
171  for (int k = 0; k < Z; k++) {
172  for (int j = 0; j < Y; j++) {
173  for (int i = 0; i < X; i++) {
174  UINT flattenZYX = k * (X * Y) + j * X + i;
175  UINT flattenXYZ = i * (Y * Z) + j * Z + k;
176  for (int c = 0; c < numComponents; c++) {
177  _data[flattenZYX * numComponents + c] = data[flattenXYZ * numComponents + c];
178  }
179  counter++;
180  }
181  }
182  }
183  std::swap(data, _data);
184  }
