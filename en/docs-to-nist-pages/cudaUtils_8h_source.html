270 
271  const Real & a = vec1[0];
272  const Real & b = vec1[1];
273  const Real & c = vec1[2];

275  const Complex & p1 = vec2[0];
276  const Complex & p2 = vec2[1];
277  const Complex & p3 = vec2[2];

279  Real res = 0;
280  Complex _temp;
281  _temp.x = -a*a*p1.x + d*p1.x - a*(b*p2.x + c*p3.x);
282  _temp.y = -a*a*p1.y + d*p1.y - a*(b*p2.y + c*p3.y);
283  res = _temp.x*_temp.x + _temp.y*_temp.y;
284 
285  _temp.x = -a*b*p1.x - b*b*p2.x + d*p2.x - b*c*p3.x;
286  _temp.y = -a*b*p1.y - b*b*p2.y + d*p2.y - b*c*p3.y;
287  res += _temp.x*_temp.x + _temp.y*_temp.y;
288 
289  _temp.x = -a*c*p1.x - b*c*p2.x + (-c*c + d)*p3.x;
290  _temp.y = -a*c*p1.y - b*c*p2.y + (-c*c + d)*p3.y;
291  res += _temp.x*_temp.x + _temp.y*_temp.y;
292 
293  return res;
