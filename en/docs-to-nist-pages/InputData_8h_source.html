130  template <typename T>
131  void ReadArrayRequired(libconfig::Config & config, const std::string& key,
132  std::vector<T>& arr, const UINT size = 0) {
133  // this is here because lookup throws an exception if not found
134  if (!config.exists(key)) {
135  std::cerr << "[Input Error] No value corresponding to " << key
136  << " found. Exiting\n";
137  exit(EXIT_FAILURE);
138  }
139 
140  libconfig::Setting &setting = config.lookup(key);
141  if (!setting.isArray()) { // confirm this is an array
142  std::cerr << "expected array input but found single value "
143  << "(key = " << key << ")\n";
144  exit(EXIT_FAILURE);
145  }
146  if(size > 0){
147  if(setting.getLength() != size){
148  std::cout << RED << "[Input Error]: The array corresponding to " << key << " must be of length " << size
149  << ". But found of length " << setting.getLength() << NRM <<"\n";
150  exit(EXIT_FAILURE);
151  }
152  }
153  arr.clear();
154  for (int i = 0; i < setting.getLength(); i++) {
155  T value;
156  std::string item_path = key + ".[" + std::to_string(i) + "]";
157  ReadValueRequired(config, item_path, value);
158  arr.push_back(value);
159  }
160  }

258  InputData(std::string filename = "config.txt") {
259  libconfig::Config cfg;
260  try {
261  cfg.readFile(filename.c_str());
262  }
263  catch (libconfig::FileIOException & e) {
264  std::cerr << " Cannot read " << filename << "\n";
265  exit(EXIT_FAILURE);
266  }catch (libconfig::ParseException &e) {
267  std::cerr << "Parse error at " << e.getFile() << ":" << e.getLine()
268  << " - " << e.getError() <<"\n";
269  exit(EXIT_FAILURE);
270  }
271  ReadValueRequired(cfg, "CaseType",caseType);
272  ReadArrayRequired(cfg, "Energies", energies);
273  std::vector<Real> _temp;
274  ReadArrayRequired(cfg, "EAngleRotation", _temp,3);
275 
276  startAngle = _temp[0]; incrementAngle = _temp[1]; endAngle = _temp[2];
277  if(startAngle > endAngle){
278  throw std::runtime_error("Start EAngle can not be greater than end EAngle");
279  }
280  if(incrementAngle == 0){
281  if(not(FEQUALS(startAngle,endAngle))){
282  throw std::runtime_error("Increment angle = 0");
283  }
284  else {
285  incrementAngle = 1.0;
286  }
287  }
288  ReadValueRequired(cfg, "MorphologyType", morphologyType);
289 
290  if(ReadValue(cfg, "NumThreads", num_threads)){}
291  if(ReadValue(cfg, "RotMask",rotMask)){}
292  if(ReadValue(cfg, "EwaldsInterpolation",ewaldsInterpolation)){}
293 // if(ReadValue(cfg, "WriteVTI",writeVTI)){}
294 // if(ReadValue(cfg, "VTIDirName",VTIDirName)){}
295  if(ReadValue(cfg, "HDF5DirName",HDF5DirName)){}
296  if(ReadValue(cfg, "WindowingType",windowingType)){}
297  if(ReadValue(cfg, "Algorithm",algorithmType)){}
298  if(ReadValue(cfg,"ScatterApproach",scatterApproach)){}
299  if(ReadValue(cfg,"DumpMorphology",dumpMorphology)){}
300  if(ReadValue(cfg,"MaxStreams",numMaxStreams)){}
301  UINT _temp1;
302  if(ReadValue(cfg,"ReferenceFrame",_temp1)){
303  validate("Reference Frame ",_temp1,2);
304  referenceFrame = static_cast<bool>(_temp1);
305  }
306 
307  if(caseType == CaseTypes::DEFAULT) {
308  kVectors.resize(1,{0,0,1});
309  }
310  else {
311  const std::string key = "listKVectors";
312  if (!cfg.exists(key)) {
313  std::cerr << "[Input Error] No value corresponding to " << key
314  << " found. Exiting\n";
315  exit(EXIT_FAILURE);
316  }
317  const libconfig::Setting & listOfKVectors = cfg.getRoot()[key];
318  kVectors.resize(listOfKVectors.getLength());
319  for(int i = 0; i < kVectors.size(); i++) {
320  const libconfig::Setting & kRoot = listOfKVectors[i].lookup("k");
321  std::array<Real,3> k;
322  ReadArray<Real,3>(kRoot,"k",k);
323  kVectors[i].x = k[0];
324  kVectors[i].y = k[1];
325  kVectors[i].z = k[2];
326  normalizeVec(kVectors[i]);
327  }
328  }
329  if(caseType == CaseTypes::GRAZING_INCIDENCE){
330  ReadArrayRequired(cfg, "DetectorCoordinates", _temp,3);
331  detectorCoordinates.x = _temp[0];
332  detectorCoordinates.y = _temp[1];
333  detectorCoordinates.z = _temp[2];
334  normalizeVec(detectorCoordinates);
335  }
336  }

338  void readRefractiveIndexData(std::vector<Material> &refractiveIndex) const {
339  libconfig::Config cfg;
340  refractiveIndex.resize(energies.size()*NUM_MATERIAL);
341  const UINT & numEnergy = energies.size();
342  for (int numMaterial = 0; numMaterial < NUM_MATERIAL; numMaterial++) {
343  std::string fname = "Material" + std::to_string(numMaterial+1) + ".txt";
344  try {
345  cfg.readFile(fname.c_str());
346  }
347  catch (libconfig::FileIOException & e) {
348  std::cerr << " Cannot read " << fname << "\n";
349  exit(EXIT_FAILURE);
350  }catch (libconfig::ParseException &e) {
351  std::cerr << "Parse error at " << e.getFile() << ":" << e.getLine()
352  << " - " << e.getError() <<"\n";
353  exit(EXIT_FAILURE);
354  }
355  for (int i = 0; i < numEnergy; i++) {
356  const auto &global = cfg.getRoot()["EnergyData" + std::to_string(i)];
357  Real energy = global["Energy"];
358  Real currEnergy =energies[i];
359  Real diff = fabs(energy - currEnergy);
360 
361  if (diff > 1E-3) {
362  std::cout << "[Input Error] No energy found for " << currEnergy << "\n";
363  exit(EXIT_FAILURE);
364  }
365 
366  Real deltaPara = global["DeltaPara"];
367  Real betaPara = global["BetaPara"];
368  Real deltaPerp = global["DeltaPerp"];
369  Real betaPerp = global["BetaPerp"];
370 
371  refractiveIndex[i*NUM_MATERIAL + numMaterial].npara.x = 1 - deltaPara;
372  refractiveIndex[i*NUM_MATERIAL + numMaterial].npara.y = betaPara;
373 
374  refractiveIndex[i*NUM_MATERIAL + numMaterial].nperp.x = 1 - deltaPerp;
375  refractiveIndex[i*NUM_MATERIAL + numMaterial].nperp.y = betaPerp;
376  }
377 
378  }
