115  void writeMorphologyFile(const std::string &fname, const InputData &inputData, const Voxel *morphologyData, const int NUM_MATERIAL) {
116  std::array<std::string, 4> morphologyDataSets{};
117  const MorphologyType morphologyType = static_cast<MorphologyType>(inputData.morphologyType);
118  if (morphologyType == MorphologyType::VECTOR_MORPHOLOGY) {
119  morphologyDataSets[0] = "Sx";
120  morphologyDataSets[1] = "Sy";
121  morphologyDataSets[2] = "Sz";
122  morphologyDataSets[3] = "UnalignedFraction";
123  } else if (morphologyType == MorphologyType::EULER_ANGLES) {
124  morphologyDataSets[0] = "S";
125  morphologyDataSets[1] = "Theta";
126  morphologyDataSets[2] = "Psi";
127  morphologyDataSets[3] = "Vfrac";
128  } else {
129  throw std::runtime_error("Unexpected Morphology Type");
130  }
131  const std::string filename = fname + ".h5";
132  H5::H5File file(filename.c_str(), H5F_ACC_TRUNC);
133  try {
134 
135  const BigUINT numVoxels = inputData.voxelDims[0] * inputData.voxelDims[1] * inputData.voxelDims[2];
136  Real *data = new Real[numVoxels];
137  const int RANK = 3;
138  const hsize_t dims[3]{inputData.voxelDims[2], inputData.voxelDims[1], inputData.voxelDims[0]}; // C++ order
139 
140  for (int numMat = 0; numMat < NUM_MATERIAL; numMat++) {
141  const auto &group = file.createGroup("Material_" + std::to_string(numMat + 1));
142  for (int numComponent = 0; numComponent < 4; numComponent++) {
143  for (int numVoxel = 0; numVoxel < numVoxels; numVoxel++) {
144  int offset = numMat*numVoxels + numVoxel ;
145  data[numVoxel] = morphologyData[offset].getValueAt(numComponent);
146  }
147  H5::DataSpace dataspace(RANK, dims);
148 #ifdef DOUBLE_PRECISION
149  H5::DataSet dataSet = group.createDataSet(morphologyDataSets[numComponent].c_str(), H5::PredType::NATIVE_DOUBLE, dataspace);
150  dataSet.write(data, H5::PredType::NATIVE_DOUBLE);
151 #else
152  H5::DataSet dataSet = group.createDataSet(morphologyDataSets[numComponent].c_str(),
153  H5::PredType::NATIVE_FLOAT, dataspace);
154  dataSet.write(data, H5::PredType::NATIVE_FLOAT);
155 #endif
156  H5DSset_label(dataSet.getId(),0,"Z");
157  H5DSset_label(dataSet.getId(),1,"Y");
158  H5DSset_label(dataSet.getId(),2,"X");
159  dataSet.close();
160  }
161  }
162  delete[] data;
163  }
