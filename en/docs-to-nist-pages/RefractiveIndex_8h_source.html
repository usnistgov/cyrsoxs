77  void addData(const std::vector<std::vector<Real>> &values, const Real Energy) {
78  enum EnergyValues : u_short {
79  DeltaPara = 0,
80  BetaPara = 1,
81  DeltaPerp = 2,
82  BetaPerp = 3
83  };
84  const int NUM_MATERIAL = inputData_.NUM_MATERIAL;
85  if (not(values.size() == NUM_MATERIAL)) {
86  py::print("Wrong input for Energy. Number not matching with number of Materials");
87  return;
88  }
89  for (auto &value:values) {
90  if ((value.size() != 4)) {
91  py::print("Wrong number of input parameters. Parameters must be in the order of "
92  "(DeltaPara, BetaPara, DeltaPerp, , BetaPerp)");
93  return;
94  }
95  }
96  UINT counter = (std::lower_bound(energies.begin(), energies.end(), Energy) - energies.begin());
97  if (not(FEQUALS(energies[counter], Energy))) {
98  py::print("Bad Energy. Trying to add parameters for non-existent energy.");
99  return;
100  }
101  for (UINT i = 0; i < NUM_MATERIAL; i++) {
102  refractiveIndex[counter*NUM_MATERIAL+i].npara.x = 1 - values[i][EnergyValues::DeltaPara];
103  refractiveIndex[counter*NUM_MATERIAL+i].npara.y = values[i][EnergyValues::BetaPara];
104  refractiveIndex[counter*NUM_MATERIAL+i].nperp.x = 1 - values[i][EnergyValues::DeltaPerp];
105  refractiveIndex[counter*NUM_MATERIAL+i].nperp.y = values[i][EnergyValues::BetaPerp];
106  }
107 
108  isValid[counter] = true;
109 
110  }
