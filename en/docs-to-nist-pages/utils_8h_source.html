38 static void writeH5(const InputData & inputData, const UINT * voxelSize,const Real *projectionGPUAveraged, const std::string dirName = "HDF5"){
39  createDirectory(dirName);
40  // TODO: Make it work in parallel.
41  omp_set_num_threads(1);
42 
43  const UINT numEnergyLevel = inputData.energies.size();
44  const UINT voxel2DSize = voxelSize[0] * voxelSize[1];
45  Real *oneEnergyData = new Real[voxel2DSize];
46  UINT chunkSize = static_cast<UINT>(std::ceil(numEnergyLevel * 1.0 / (omp_get_num_threads() * 1.0)));
47  UINT threadID = omp_get_thread_num();
48  UINT startID = threadID * chunkSize;
49  UINT endID = ((threadID + 1) * chunkSize);
50 
51  for (UINT csize = startID; csize < std::min(endID, numEnergyLevel); csize++) {
52 
53  std::stringstream stream;
54  Real energy = inputData.energies[csize];
55  stream << std::fixed << std::setprecision(2) << energy;
56  std::string s = stream.str();
57  const std::string outputFname = dirName + "/Energy_" + s + ".h5";
58  H5::H5File file(outputFname.c_str(), H5F_ACC_TRUNC);
59  {
60  H5::Group group(file.createGroup("KIDList"));
61  std::vector<std::array<Real, 3>> _kList(inputData.kVectors.size());
62  for (int i = 0; i < _kList.size(); i++) {
63  _kList[i] = {inputData.kVectors[i].x, inputData.kVectors[i].y, inputData.kVectors[i].z};
64  }
65  const hsize_t dims[2]{inputData.kVectors.size(), 3};
66  const int RANK = 2;
67  H5::DataSpace dataspace(RANK, dims);
68  H5::DataSet dataset = group.createDataSet("KVec", H5::PredType::NATIVE_FLOAT, dataspace);
69 #ifdef DOUBLE_PRECISION
70  dataset.write(_kList.data(), H5::PredType::NATIVE_DOUBLE);
71 #else
72  dataset.write(_kList.data(), H5::PredType::NATIVE_FLOAT);
73 #endif
74  group.close();
75  }
76  for (UINT kID = 0; kID < inputData.kVectors.size(); kID++) {
77  const std::size_t offset = static_cast<std::size_t>(csize) * static_cast<std::size_t>(voxel2DSize) * inputData.kVectors.size() +
78  static_cast<std::size_t>(kID*voxel2DSize);
79  std::memcpy(oneEnergyData, &projectionGPUAveraged[offset], sizeof(Real) * voxel2DSize);
80  const std::string groupname = "K" + std::to_string(kID);
81  H5::writeFile2D(file, oneEnergyData, voxelSize,groupname);
82  }
83  file.close();
84  }
85 
86  delete[] oneEnergyData;
87 }
